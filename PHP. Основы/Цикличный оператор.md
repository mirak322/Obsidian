foreach

foreach($person as $item) -  тут начинается цикл оператор - в скобках задаются аргументы

персон - массив где есть ключ и значаен
хотим представить в виде переменной item
foreach($person as $item) {
	echo $item;
}

слева стопт кей справа айтем:

foreach($person as $key => $item) {
	echo $item;
}


foreach($person as $key => $item) {
	echo $key . ': ';
	echo $item . "\n";
}
